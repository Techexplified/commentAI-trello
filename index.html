const ICON = 'https://cdn-icons-png.flaticon.com/512/2103/2103633.png'; window.TrelloPowerUp.initialize({ /* CARD BUTTON (Toggle) */ 'card-buttons': function(t, options) { return t.get('card', 'shared', 'commentAIVisible', false) .then(function(isVisible){ return [{ icon: ICON, text: isVisible ? 'Remove Comment AI' : 'Add Comment AI', callback: function(t){ return t.set('card', 'shared', 'commentAIVisible', !isVisible) .then(function(){ return t.reload(); // refresh card UI }); } }]; }); }, /* CARD SECTION */ 'card-back-section': function(t, options){ return t.get('card','shared','commentAIVisible',false) .then(function(isVisible){ if(!isVisible){ return null; // hide section } return { title: 'Comment AI', icon: ICON, content: { type: 'iframe', url: t.signUrl('./section.html'), height: 400 } }; }); }, /* SETTINGS (only for board menu) */ 'show-settings': function(t, options) { return t.popup({ title: 'Comment AI Settings', url: './settings.html', height: 250 }); } });
